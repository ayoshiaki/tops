55  Sequence & DecodableModel::choose(Sequence & h, Sequence & path, int i, int size) const
56  {
57  assert(path.size() == h.size());
58  int state = chooseFirstState();
59  if (((i - 1) < (int) path.size()) && (i - 1) >= 0)
60  state = path[i - 1];
61  int l = 0;
62  while ((int) h.size() < (i + size)) {
63  chooseObservation(h, i + l, state);
64  for (int k = i + l; k < (int) h.size(); k++, l++) {
65  if (k < (int) path.size())
66  path[k] = state;
67  else
68  path.push_back(state);
69  }
70  state = chooseState(state); // next state
71  }
72  h.resize(size);
73  path.resize(size);
74  return h;
75  }

96  void DecodableModel::posteriorProbabilities (const Sequence &sequence, Matrix & probabilities) const
97  {
98  int nstates = (int)getStateNames()->size();
99  int size = sequence.size();
100  Matrix p (nstates, size);
101 
102  Matrix alpha; // forward
103  Matrix beta; // backward
104 
105  double full = forward(sequence, alpha);
106  backward(sequence, beta);
107 
108  for(int k = 0; k < nstates; k++)
109  for(int i = 0; i < size; i++)
110  p(k, i) = alpha(k, i) + beta(k, i) - full;
111 
112  probabilities = p;
113  }

117  void DecodableModel::posteriorDecoding (const Sequence &sequence, Sequence &path, Matrix & probabilities) const{
118  int nstates = (int)getStateNames()->size();
119  int size = sequence.size();
120 
121  posteriorProbabilities(sequence, probabilities);
122 
123  path.resize(size);
124 
125  double max;
126  for(int i = 0; i < size; i++)
127  {
128  max = probabilities(0, i);
129  path[i] = 0;
130  for(int k = 1; k < nstates; k++)
131  {
132  if(probabilities(k, i) > max)
133  {
134  max = probabilities(k, i);
