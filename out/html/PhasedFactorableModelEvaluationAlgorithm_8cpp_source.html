81  {
82  _alpha[k].resize(s.size() +1);
83  _precision[k].resize(s.size()+1);
84  }
85  for(int k = 0; k < nphases; k++)
86  {
87  _alpha[k][0] = 0;
88  _precision[k][0] = 0;
89  history.resize(0);
90  for(int i = 0; i < (int)s.size(); i++)
91  {
92  double prob = m->log_probability_of (history, s, i, k);
