86 void TrainProfileHMMMaxLikelihood::infer_model(string fastafile, int pseudocounts, double residuefraction, string datetime, const vector<string>& symbols) const {
87  string line, sequencia;
88  vector<string> seq;
89  vector<string> seq_names;
90 
91  ifstream myfile(fastafile.c_str());
92 
93  if (myfile.is_open()) {
94  while (!myfile.eof()) {
95  getline(myfile, line);
96  if (match_expression(line, ">.*")) {
97  if (seq_names.size() != 0) {
98  seq.push_back(sequencia);
99  sequencia = "";
100  }
101  seq_names.push_back(line.substr(1, line.length()));
102  } else {
103  sequencia += line;
104  }
105  }
106  if (seq_names.size() != 0) {
107  seq.push_back(sequencia);
108  sequencia = "";
109  }
110  myfile.close();
111 
112  }
113 
114  int count[seq[0].size()];
115  bool match[seq[0].size()];
116  int index[seq[0].size()];
117  int pos = 0;
118  int n_match_states = 0;
119 
120  for (unsigned int z = 0; z < seq[0].size(); z++) {
121  count[z] = 0;
122  }
123  pos = 0;
124  for (unsigned int j = 0; j < seq[0].size(); j++) {
125  unsigned int i;
126  for (i = 0; i < seq.size(); i++) {
127  if (seq[i][j] != '.' && seq[i][j] != '-') {
128  count[j]++;
129  }
130  }
131  if (count[j] > seq.size() * residuefraction) {
132  match[j] = true;
133  index[j] = ++pos;
134  n_match_states++;
135  } else {
136  match[j] = false;
137  index[j] = pos;
138  }
139  }
140  string emiss_filename = ".emiss_" + datetime + ".txt";
141  string states_filename = ".states_" + datetime + ".txt";
142  string model_filename = ".model_" + datetime + ".txt";
143  string config_filename = ".config_" + datetime + ".txt";
144 
145  config_file_creator(states_filename, emiss_filename, model_filename, config_filename, pseudocounts);
146  model_file_creator(n_match_states, symbols, model_filename);
147  emiss_states_file_creator(seq, index, match, seq_names, emiss_filename, states_filename);
148 
149 }

158  ofstream model;

160  model.open(model_filename.c_str());

162  model << "state_names = (\"M0\"";
163 
164  for (int i = 1; i < n_match_states + 2; i++) {
165  model << ",\"M" << i << "\"";
166  }
167  for (int i = 0; i < n_match_states + 1; i++) {

170  for (int i = 1; i < n_match_states + 1; i++) {
171  model << ",\"D" << i << "\"";
172  }
173  model << ")" << endl;
174 
175  model << "observation_symbols = (";
176  string str = "";
177  for (unsigned int i = 0; i < symbols.size(); i++) {
178  str += "\"" + symbols[i] + "\",";
179  }
180  model << str.substr(0, str.length() - 1) << ")" << endl;
181 
182  model << "transitions = (\"I0\" | \"M0\": 0.33333;" << endl;
183  model << "\t\t\"D1\" | \"M0\": 0.33333;" << endl;
184  model << "\t\t\"M1\" | \"M0\": 0.33333;" << endl;
185  model << "\t\t\"D1\" | \"I0\": 0.33333;" << endl;
186  model << "\t\t\"I0\" | \"I0\": 0.33333;" << endl;
187  model << "\t\t\"M1\" | \"I0\": 0.33333;" << endl;
188 
189  for (int i = 1; i < n_match_states; i++) {
190  int next = i + 1;
191  model << "\t\t\"I" << i << "\" | " << "\"M" << i << "\": 0.33333;" << endl;
192  model << "\t\t\"D" << next << "\" | " << "\"M" << i << "\": 0.33333;" << endl;
193  model << "\t\t\"M" << next << "\" | " << "\"M" << i << "\": 0.33333;" << endl;
194  model << "\t\t\"D" << next << "\" | " << "\"I" << i << "\": 0.33333;" << endl;
195  model << "\t\t\"I" << i << "\" | " << "\"I" << i << "\": 0.33333;" << endl;
196  model << "\t\t\"M" << next << "\" | " << "\"I" << i << "\": 0.33333;" << endl;
197  model << "\t\t\"D" << next << "\" | " << "\"D" << i << "\": 0.33333;" << endl;
198  model << "\t\t\"I" << i << "\" | " << "\"D" << i << "\": 0.33333;" << endl;
199  model << "\t\t\"M" << next << "\" | " << "\"D" << i << "\": 0.33333;" << endl;
200  }
201 
202  int last = n_match_states + 1;
203  int prev = n_match_states;
204 
205  model << "\t\t\"I" << prev << "\" | " << "\"M" << prev << "\": 0.5;" << endl;
206  model << "\t\t\"M" << last << "\" | " << "\"M" << prev << "\": 0.5;" << endl;
207  model << "\t\t\"I" << prev << "\" | " << "\"I" << prev << "\": 0.5;" << endl;
208  model << "\t\t\"M" << last << "\" | " << "\"I" << prev << "\": 0.5;" << endl;
209  model << "\t\t\"I" << prev << "\" | " << "\"D" << prev << "\": 0.5;" << endl;
210  model << "\t\t\"M" << last << "\" | " << "\"D" << prev << "\": 0.5;" << endl;
211  model << "\t\t\"M" << last << "\" | " << "\"M" << last << "\": 1.0)" << endl;
212 
213  long double prob = 1.0 / symbols.size();
214  model << "emission_probabilities = (";
215  string state = "I0";
216  for (unsigned int j = 0; j < symbols.size(); j++) {
217  char sbl = symbols[j][0];

221  for (unsigned int j = 0; j < symbols.size(); j++) {
222  char sbl = symbols[j][0];
223  model << "\"" << sbl << "\" | \"" << "M" << i << "\": " << setprecision(10) << prob << ";" << endl;
224  }
225  for (unsigned int j = 0; j < symbols.size(); j++) {
226  char sbl = symbols[j][0];
227  if (j < symbols.size() - 1 || i < n_match_states) {
228  model << "\"" << sbl << "\" | \"" << "I" << i << "\": " << setprecision(10) << prob << ";" << endl;
229  } else {
230  model << "\"" << sbl << "\" | \"" << "I" << i << "\": " << setprecision(10) << prob << ")" << endl;
231  }
232  }
233  }
234  model << "initial_probabilities= (\"M0\": 1.0)" << endl;

252 void TrainProfileHMMMaxLikelihood::emiss_states_file_creator(const vector<string>& seq, int index[], bool match[], const vector<string>& seq_names, string emiss_filename,
253  string states_filename) const {
254  ofstream emiss, states;
255  emiss.open(emiss_filename.c_str());
256  states.open(states_filename.c_str());
257  for (unsigned int i = 0; i < seq.size(); i++) {
258  emiss << seq_names[i] << "_emiss: ";
259  states << seq_names[i] << "_states: ";
260  states << "M0 ";
261  for (unsigned int j = 0; j < seq[i].size(); j++) {
262  if (match[j]) {
263  if (seq[i][j] == '.' || seq[i][j] == '-') {
264  states << "D" << index[j] << " ";
265  } else {
266  emiss << seq[i][j] << " ";
267  states << "M" << index[j] << " ";
268  }
269  } else {
270  if (!(seq[i][j] == '.' || seq[i][j] == '-')) {
271  emiss << seq[i][j] << " ";
272  states << "I" << index[j] << " ";
273  }
274  }
275  }
276  states << "M" << index[seq[0].size() - 1] + 1;
277  emiss << endl;
278  states << endl;
279  }
280  emiss.close();
281  states.close();
282 }
