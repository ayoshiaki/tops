36  DiscreteIIDModel:: DiscreteIIDModel(const DoubleVector & probabilities)
37  {
38  _size = probabilities.size() - 1;
39  _log_probabilities.resize(_size+1);
40  double sum = 0;
41  for(unsigned int i = 0; i < probabilities.size(); i++)
42  sum += probabilities[i];
43  for(int i = 0; (i <= _size); i++)
44  if(i >= (int)probabilities.size())
45  _log_probabilities[i] = -HUGE;
46  else if(close(probabilities[i], 0.0, 1e-10))
47  _log_probabilities[i] = -HUGE;
48  else
49  _log_probabilities[i] = log(probabilities[i]/sum);
50  _geometric_tail = false;
51  }

128  double DiscreteIIDModel:: choose() const {
129  double random = ((double)rand())/ ((double) RAND_MAX + 1.0) ;
130  int result;
131  for(result = 0; result < (int)_log_probabilities.size(); result++)
132  {
133  random -= exp(_log_probabilities[result]);
134  if(random <= 0)
135  break;
136  }
137  if(result >= (int)_log_probabilities.size())
138  return (double)(result-1);
139  return (double)result;

142  void DiscreteIIDModel::choosePair(int* a, int* b) const {
143  double random = ((double)rand())/ ((double) RAND_MAX + 1.0) ;
144  int result1 = 0;
145  int result2 = 0;
146  for(result1 = 0; result1 < (int)_log_probabilities_matrix.size1(); result1++){
147  for(result2 = 0; result2 < (int)_log_probabilities_matrix.size2(); result2++){
148  random -= exp(_log_probabilities_matrix(result1,result2));
149  if(random <= 0)
150  break;
151  }
152  if(random <= 0)
153  break;
154  }
155  if(result1 >= (int)_log_probabilities_matrix.size1() && result2 >= (int)_log_probabilities_matrix.size2()){
156  *a = (result1 - 1);
157  *b = (result2 - 1);
158  }
159  else{
160  *a = result1;
161  *b = result2;

205  void DiscreteIIDModel::initializeFromMap(const std::map <std::string, double> & probabilities, AlphabetPtr alphabet)
206  {
207  std::map<std::string,double>::const_iterator it;
208  DoubleVector probs;
209  probs.resize(alphabet->size());
210  for(int i = 0; i < (int)alphabet->size() ; i++)
211  probs[i] = 0.0;
212  for(it = probabilities.begin(); it != probabilities.end(); it++)
213  {
214 
215  int id = alphabet->getSymbol(it->first)->id();
216  if(id < (int)probs.size())
217  probs[id] = it->second;
218  }
219  _log_probabilities.resize(probs.size());
220  double sum = 0;
221  for(unsigned int i = 0; i < probs.size(); i++)
222  {
223  sum += probs[i] ;
224  if(close(probs[i], 0.0, 1e-100))
225  _log_probabilities[i] = -HUGE;
226  else
227  _log_probabilities[i] = log(probs[i]);
228  }
229  if(close(sum, 1.0, 1e-5))
230  _log_probabilities.push_back(-HUGE);
231  else {
232  _log_probabilities[probs.size()-1] = log(1.0 - sum);
233  std::cerr << "WARNING: DiscreteIIDModel parameters must sum 1.0: sum is " << sum << std::endl;
234  }
235  setAlphabet(alphabet);
236  }

240  ProbabilisticModelParameterValuePtr alphabet = p.getOptionalParameterValue("alphabet");
241  ProbabilisticModelParameterValuePtr geometric_tail_ptr = p.getOptionalParameterValue("geometric_tail");
242  DoubleVector distr = probs->getDoubleVector();
243  setProbabilities(distr);
244  if (alphabet != NULL)

246  AlphabetPtr alpha = AlphabetPtr(new Alphabet());
247  alpha->initializeFromVector(alphabet->getStringVector());
248  setAlphabet(alpha);
249  if(geometric_tail_ptr != NULL) {
250  _geometric_tail = (geometric_tail_ptr->getInt() == 1);
251  }
252  }
253  _mean = 0.0;
254  for (int i = 0 ; i < _log_probabilities.size(); i++){
255  _mean += exp(_log_probabilities[i]) * i;
