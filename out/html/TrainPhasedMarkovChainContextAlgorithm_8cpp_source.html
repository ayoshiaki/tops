31 ProbabilisticModelPtr TrainPhasedMarkovChainContextAlgorithm::create(
32  ProbabilisticModelParameters & parameters, double & loglikelihood,
33  int & sample_size) const {
34  ProbabilisticModelParameterValuePtr alphabetpar =
35  parameters.getMandatoryParameterValue("alphabet");
36  ProbabilisticModelParameterValuePtr trainingsetpar =
37  parameters.getMandatoryParameterValue("training_set");
38  ProbabilisticModelParameterValuePtr cutpar =
39  parameters.getMandatoryParameterValue("cut");
40 
41  ProbabilisticModelParameterValuePtr number_of_phasespar =
42  parameters.getMandatoryParameterValue("number_of_phases");
43 
44  if ((alphabetpar == NULL) || (trainingsetpar == NULL)
45  || (number_of_phasespar == NULL)
46  || (cutpar == NULL)) {
47  std::cerr << help() << std::endl;
48  exit(-1);
49  }
50  AlphabetPtr alphabet = AlphabetPtr(new Alphabet());
51  alphabet->initializeFromVector(alphabetpar ->getStringVector());
52  SequenceEntryList sample_set;
53  readSequencesFromFile(sample_set, alphabet, trainingsetpar->getString());
54  int length = number_of_phasespar->getInt() ;
55  std::vector<ContextTreePtr> positional_distribution;
56  positional_distribution.resize(length);
57  sample_size = 0;
58  std::vector <std::string> cut = cutpar->getStringVector();
59  if((int) cut.size() < length) {
60  std::cerr << "ERROR: The number of cut values is smaller than the sequence length " << std::endl;
61  exit(-1);
62  }
63 
64  for (int i = 0; i < length; i++) {
65  SequenceEntryList positionalSample;
66  for(int j = 0; j < (int)sample_set.size(); j++)
67  {
68  int nseq = 0;
69  int order = j;
70  if(order > 10)
71  order = 10;
72  while(true)
73  {
74  int start = (length) * nseq - order + i;
75  if(start < 0) {
76  nseq++;
77  continue;
78  }
79  int end = (length) * nseq + i;
80  if(end >= (int)(sample_set[j]->getSequence()).size())
81  break;
82  Sequence s;
83  for(int k = start; k <= end; k++)
84  s.push_back((sample_set[j]->getSequence())[k]);
85  SequenceEntryPtr entry = SequenceEntryPtr(new SequenceEntry(alphabet));
86  entry->setSequence(s);
87  positionalSample.push_back(entry);
88  nseq++;
89  }
90  }
91  double delta = (parameters.getOptionalParameterValue(cut[i])->getDouble());
92  ContextTreePtr tree = ContextTreePtr(new ContextTree(alphabet));
93  tree->initializeContextTreeRissanen(positionalSample);
94  tree->pruneTree(delta);
95  tree->removeContextNotUsed();
96  tree->normalize();
97  positional_distribution[i] = tree;
98  }
99  InhomogeneousMarkovChainPtr model = InhomogeneousMarkovChainPtr(
100  new InhomogeneousMarkovChain());
101  model->setPositionSpecificDistribution(positional_distribution);
102  model->phased(1);
103  model->setAlphabet(alphabet);
104  loglikelihood = 0.0;
105 
106  for (int j = 0; j < (int) sample_set.size(); j++) {
107  sample_size += (sample_set[j]->getSequence()).size();
108  loglikelihood += model->evaluate((sample_set[j]->getSequence()), 0, (sample_set[j]->getSequence()).size()
109  - 1);
110  }
111  return model;
112 
113 }
