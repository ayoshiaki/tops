37  void restore_submodel(std::string & model_name, std::map<std::string, ProbabilisticModelPtr> &_models, const ProbabilisticModelParameters & parameters) {
38  ProbabilisticModelParameterValuePtr modelpar =
39  parameters.getOptionalParameterValue(model_name);
40 
41  if (modelpar == NULL) {
42  std::cerr << "ERROR:: Missing definition of the model "
43  << model_name << std::endl;
44  return;
45  }
46  if (_models.find(model_name) != _models.end()) {
47  return;
48  }
49 
50  std::string cfg = modelpar->getString();
51  ProbabilisticModelCreatorClient creator;
52  ConfigurationReader modelreader;
53  if ((cfg.size()) > 0 && (cfg[0] == '[')) {
54  cfg = cfg.substr(1, cfg.size() - 2);
55  if (modelreader.load(cfg)) {
56  ProbabilisticModelParameterValuePtr modelpar = (modelreader.parameters())->getOptionalParameterValue("model");
57  if(modelpar != NULL)
58  {
59  std::string submodelstr = modelpar->getString();
60  if (!((submodelstr.size()) > 0 && (submodelstr[0] == '['))) {
61  restore_submodel(submodelstr, _models, parameters);
62  }
63  }
64  ProbabilisticModelPtr m = creator.create(*(modelreader.parameters()), _models);
65  _models[model_name] = m;
66  } else{
67  std::cerr << "/=======/\n" << cfg << "/========/" << std::endl;
68  exit(-1);
69  }
70  }
71  else
72  {
73  ProbabilisticModelPtr m = creator.create(cfg);
74  if (m == NULL) {
75  std::cerr << "Can not load model " << cfg << "! " << std::endl;
76  return;
77  }
78  _models[model_name] = m;
79  }
80  }

83  double MultipleSequentialModel::prefix_sum_array_compute(int begin, int end, int phase) {
84  if(begin < 0)
85  return -HUGE;
86  if(begin > end)
87  return -HUGE;
88  double sum = 0;
89  int b = begin;
90  int e = 0;
91  for(int i = 0; i < _idx_not_limited; i++)
92  {
93  e = b + _max_size[i] - 1;
94  if (e >= _seqsize)
95  e = _seqsize-1;
96  sum += _sub_models[i]->prefix_sum_array_compute(b,e,phase);
97  if( e >= (int)end)
98  return sum;
99 
100  phase = mod(phase + e - b + 1, 3);
101  b = e + 1;
102  if(e>=_seqsize)
103  break;
104  }
105  int begin_of_not_limited = b;
106  e = end;
107  for (int i = _sub_models.size()-1; i > _idx_not_limited ; i--)
108  {
109  b = e - _max_size[i] + 1;
110  int phase2 = mod(phase + b - begin_of_not_limited, 3);
111  if(b < 0) {
112  phase2 = mod(phase2 -b, 3);
113  b = 0;
114  }
115  sum += _sub_models[i]->prefix_sum_array_compute(b,e,phase2);
116  e = b - 1;
117  if (e < 0)
118  break;
119  }
120  int end_of_not_limited = e;
121  if( end_of_not_limited - begin_of_not_limited + 1 > 0 ){
122  sum += _sub_models[_idx_not_limited]->prefix_sum_array_compute(begin_of_not_limited, end_of_not_limited, phase);
123  }
124  return sum;
125  }

204  void MultipleSequentialModel::initialize (const ProbabilisticModelParameters & p )
205  {
206  ProbabilisticModelParameterValuePtr modelspar = p.getMandatoryParameterValue("models");
207  ProbabilisticModelParameterValuePtr maxsizepar = p.getMandatoryParameterValue("max_length");
208  ProbabilisticModelCreatorClient creator;
209  StringVector modelnames = modelspar->getStringVector();
210  DoubleVector maxlength = maxsizepar->getDoubleVector();
211 
212  if(maxlength.size() != modelnames.size() )
213  {
214  std::cerr << "ERROR: number of models does not match the number of max_length";
215  exit(-1);
216  }
217  _sub_models.resize(modelnames.size());
218  _max_size.resize(modelnames.size());
219  _idx_not_limited = _sub_models.size() - 1;
220  int count = 0;
221  for(int i = 0; i < (int)modelnames.size();i++)
222  {
223  if(maxlength[i] < 0)
224  {
225 
226  count++;
227  _idx_not_limited = i;
228  }
229  }
230  if (count > 1)
231  {
232  std::cerr << "ERROR: Only one model can has unlimited length\n" << std::endl;
233  exit(-1);
234  }
235  for(int i = 0; i < (int)modelnames.size();i++)
236  {
237  restore_submodel(modelnames[i], _models, p);
238  _sub_models[i] = _models[modelnames[i]];
239  if(_sub_models[i] == NULL)
240  {
241  std::cerr << "ERROR: Cannot load model " << modelnames[i] << std::endl;
242  exit(-1);
243  }
244  setAlphabet(_sub_models[i]->alphabet());
245  _max_size[i] = maxlength[i];
246  }
247  _parameters = p;
248  }
