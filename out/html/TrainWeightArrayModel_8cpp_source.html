33  ProbabilisticModelPtr TrainWeightArrayModel::create( ProbabilisticModelParameters & parameters, double & loglikelihood, int & sample_size) const
34  {
35  ProbabilisticModelParameterValuePtr alphabetpar = parameters.getMandatoryParameterValue("alphabet");
36  ProbabilisticModelParameterValuePtr trainingsetpar = parameters.getMandatoryParameterValue("training_set");
37  ProbabilisticModelParameterValuePtr orderpar = parameters.getMandatoryParameterValue("order");
38  ProbabilisticModelParameterValuePtr lengthpar = parameters.getMandatoryParameterValue("length");
39  ProbabilisticModelParameterValuePtr offsetpar = parameters.getOptionalParameterValue("offset");
40  ProbabilisticModelParameterValuePtr vicinitypar = parameters.getOptionalParameterValue("vicinity_length");
41  ProbabilisticModelParameterValuePtr pseudocountspar = parameters.getOptionalParameterValue("pseudo_counts");
42  ProbabilisticModelParameterValuePtr fixseqpar = parameters.getOptionalParameterValue("fixed_sequence");
43  ProbabilisticModelParameterValuePtr fixseqpospar = parameters.getOptionalParameterValue("fixed_sequence_position");
44  ProbabilisticModelParameterValuePtr aprioripar = parameters.getOptionalParameterValue("apriori");
45 
46  ProbabilisticModelParameterValuePtr weightspar = parameters.getOptionalParameterValue("weights");
47  std::map <std::string, double> weights;
48  if(weightspar != NULL) {
49  readMapFromFile(weights, weightspar->getString());
50  }
51 
52 
53  if((alphabetpar == NULL) ||
54  (trainingsetpar == NULL) ||
55  (lengthpar == NULL) ||
56  (orderpar == NULL) )
57  {
58  std::cerr << help() << std::endl;
59  exit(-1);
60  }
61  int offset = 0;
62  double pseudocounts = 0;
63  if(pseudocountspar != NULL)
64  pseudocounts = pseudocountspar->getDouble();
65  ProbabilisticModelPtr apriori;
66  if(aprioripar != NULL)
67  {
68  ProbabilisticModelCreatorClient c;
69  apriori = c.create(aprioripar->getString());
70  }
71 
72  if(offsetpar != NULL)
73  offset = offsetpar ->getInt();
74  if(pseudocountspar != NULL)
75  pseudocounts = pseudocountspar->getDouble();
76 
77  AlphabetPtr alphabet = AlphabetPtr(new Alphabet());
78  alphabet->initializeFromVector(alphabetpar ->getStringVector());
79  SequenceEntryList sample_set;
80  readSequencesFromFile(sample_set, alphabet, trainingsetpar->getString());
81  int vicinity = 0;
82  if( vicinitypar != NULL)
83  vicinity = vicinitypar->getInt();
84  int order = orderpar->getInt();
85  int length = lengthpar->getInt();
86  std::vector <ContextTreePtr> positional_distribution;
87  positional_distribution.resize(length);
88  sample_size = 0;
89  bool fixseq = false;
90  Sequence fixed;
91  int fixed_pos = 0;
92  if((fixseqpar != NULL) && (fixseqpospar != NULL)) {
93  std::string seqstr = fixseqpar->getString();
94  SequenceFactory factory(alphabet);
95  fixed = factory.createSequence(seqstr);
96  fixed_pos = fixseqpospar->getInt();
97  fixseq = true;
98  }
99 
100  for(int i = 0; i < length; i++)
101  {
102  SequenceEntryList positionalSample;
103  int o = i;
104  if(o > order) o = order;
105  for(int j = 0; j < (int)sample_set.size(); j++)
106  {
107  for(int k = -vicinity; k <= vicinity; k++)
108  {
109  Sequence s;
110  s.resize(o+1);
111  int l = 0;
112  int m = i - o + k+ offset;
113 
114  if(m < 0)
115  continue;
116  if((m + o) >= (int)(sample_set[j]->getSequence()).size())
117  continue;
118 
119 
120  if(fixseq) {
121  while( (m < (int) (sample_set[j]->getSequence()).size()) && (l <= o))
122  {
123  s[l] = (sample_set[j]->getSequence())[m];
124  if(fixseq && (fixed_pos <= (m-k)) && ( (m-k) <= fixed_pos + fixed.size()-1)){
125  int p = m-fixed_pos-k;
126  if((p >= 0) && (p < fixed.size()))
127  s[l] = fixed[p];
128  }
129  l++; m++;
130  }
131  } else {
132  while( (m < (int) (sample_set[j]->getSequence()).size()) && (l <= o))
133  {
134  s[l] = (sample_set[j]->getSequence())[m];
135  l++; m++;
136  }
137  }
138 
139  SequenceEntryPtr entry = SequenceEntryPtr(new SequenceEntry(alphabet));
140  entry->setSequence(s);
141  entry->setName(sample_set[j]->getName());
142  positionalSample.push_back(entry);
143  }
144  }
145  if(fixseq && (fixed_pos <= i) && ((int)i <= ((int)fixed_pos + (int)fixed.size() - 1))){
146  ContextTreePtr tree = ContextTreePtr(new ContextTree(alphabet));
147  tree->initializeCounter(positionalSample, o, 0, weights);
148  tree->normalize();
149  positional_distribution[i] = tree;
150  } else {
151  ContextTreePtr tree = ContextTreePtr(new ContextTree(alphabet));
152 
153  if(apriori != NULL && apriori->factorable() != NULL){
154  tree->initializeCounter(positionalSample, o, pseudocounts, weights);
155  tree->normalize(apriori, pseudocounts);
156  } else {
157  tree->initializeCounter(positionalSample, o, pseudocounts, weights);
158  tree->normalize();
159 
160  }
161  positional_distribution[i] = tree;
162  }
163 
164  }
165  InhomogeneousMarkovChainPtr model = InhomogeneousMarkovChainPtr(new InhomogeneousMarkovChain());
166  model->setPositionSpecificDistribution(positional_distribution);
167  model->phased(0);
168  model->setAlphabet(alphabet);
169  loglikelihood = 0.0;
170 
171  for(int j = 0; j < (int)sample_set.size(); j++)
172  {
173  sample_size += (sample_set[j]->getSequence()).size();
174  double v = model->evaluate((sample_set[j]->getSequence()), 0, (sample_set[j]->getSequence()).size()-1);
175  if (v <= -HUGE)
176  continue;
177 
178  loglikelihood += v;
179  }
180  return model;
181 
182  }
