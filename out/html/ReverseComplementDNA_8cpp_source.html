32  void ReverseComplementDNA::revcomp(Sequence & revCompSeq, const Sequence & s, int begin, int end) const{
33  AlphabetPtr a = subModel()->alphabet();
34  if(end >= (int)s.size())
35  end = s.size()-1;
36  if(begin >= (int)s.size())
37  begin = s.size()-1;
38 
39  for(int i = end; i >= begin; i--) {
40  SymbolPtr symbol1 = a->getSymbol(s[i]);
41  SymbolPtr revSymbol = revAlphabet->getSymbol(symbol1->name());
42  revCompSeq.push_back(revSymbol->id());
43  }
44  }
