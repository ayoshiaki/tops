34  double FixedSequenceAtPosition::evaluate(const Sequence & s, unsigned int begin, unsigned int end) const{
35  double result = ProbabilisticModelDecorator::evaluate(s, begin, end);
36  int j;
37  for(j = 0; (j < (int)_sequence.size()) && ((_position + j) < (int)s.size()); j++)
38  if(_sequence[j] != s[_position + j] )
39  break;
40  if(j != (int)_sequence.size())
41  result += _probabilities->log_probability_of(1);
42  else
43  result += _probabilities ->log_probability_of(0);
44  return result;
45  }
46  void FixedSequenceAtPosition::addSequence(Sequence &h) const {
47  if ((int)_probabilities->choose() == 1)
48  return;
49  for(int i = _position; ((i-_position) < (int)_sequence.size()) && (i < (int)h.size()); i++)
50  h[i] = _sequence[i-_position];
51  }
