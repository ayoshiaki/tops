37  double MaximumDependenceDecomposition::evaluate(const Sequence & s, unsigned int begin, unsigned int end) const
38  {
39  double result = 0.0;
40  std::stringstream node;
41  std::vector <int> rest;
42  for(int i = 0; i < _max_size; i++){
43  rest.push_back(0);
44  }
45  node << "x";
46  while(1) {
47  std::string node_id = node.str();
48  int pos = _positions.find(node_id)->second;
49  result += (_submodels.find(node_id)->second)->inhomogeneous()->evaluatePosition(s, pos, pos);
50  rest[pos] = 1;
51  std::string last_node = node_id;
52  if(s[pos] == _consensus.find(node_id)->second) {
53  node << "l";
54  } else {
55  node << "r";
56  }
57  // if we reach the end of the tree, calculate the probabilities using the last model
58  if(_positions.find(node.str()) == _positions.end()) {
59  for(int r = 0; r < _max_size; r++)
60  if(rest[r] == 0) {
61  result += _submodels.find(last_node)->second->inhomogeneous()->evaluatePosition(s, r, r);
62  }
63  break;
64  }
65  }
66  return result;
67  }

100  void MaximumDependenceDecomposition::initialize(const ProbabilisticModelParameters & p) {
101  ProbabilisticModelParameterValuePtr maxsizepar = p.getMandatoryParameterValue("max_length");
102  ProbabilisticModelParameterValuePtr consensuspar = p.getMandatoryParameterValue("consensus");
103  ProbabilisticModelParameterValuePtr positionspar = p.getMandatoryParameterValue("positions");
104  _max_size = maxsizepar->getInt();
105  std::map<std::string,double> aux = consensuspar->getDoubleMap();
106  std::map<std::string,double>::const_iterator it;
107  for (it = aux.begin(); it != aux.end(); it++) {
108  _consensus [it->first] = (int)it->second;
109  }
110  aux = positionspar->getDoubleMap();
111  for (it = aux.begin(); it != aux.end(); it++) {
112  _positions [it->first] = (int)it->second;
113  }
114  for (it = aux.begin(); it != aux.end(); it++) {
115  std::string node = it->first;
116  restore_submodel(node, _submodels, p);
117  }
118  _parameters = p;
119  }
