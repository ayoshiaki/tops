37  ProbabilisticModelParameterValuePtr cpar =
38  parameters.getMandatoryParameterValue("C");
39  ProbabilisticModelParameterValuePtr maxlengthp =
40  parameters.getOptionalParameterValue("max_length");
41  long max = 15000;
42  if(maxlengthp != NULL)
43  max = maxlengthp->getInt();
44 
45  if((training_set_parameter == NULL)||(cpar == NULL)) {
46  std::cerr << help () << std::endl;
47  ProbabilisticModelPtr nullmodel;
48  exit(-1);
49  return nullmodel;
50  }
51  double C = cpar->getDouble();
52  DoubleVector data;
53  AlphabetPtr alpha = AlphabetPtr(new Alphabet());;
54  SequenceList sample_set;
55  readSequencesFromFile(sample_set, alpha, training_set_parameter->getString());
56 
57  for(int i = 0; i < (int)sample_set.size();i++)
58  for(int j = 0; j < (int) sample_set[i].size(); j++)
59  data.push_back(sample_set[i][j]);
60 
61  typedef std::map<long,double> Lengths;
62  int N = data.size();
63  Lengths counter;
64  Lengths sum;
65  Lengths::const_iterator iter;
66  for(int i = 0; i < N; i++){
67  iter = counter.find((long)data[i]);
68  if(iter == counter.end())
69  counter[(long)data[i]] = 1.0;
70  else
71  counter[(long)data[i]] += 1.0;
72  }
73  double total = 0.0;
74  for (int k = 1; k <= max; k++){
75  int start = k - 10;
76  int end = k + 10;
77  if(start < 0) start = 0;
78  sum[k] = 0.0;
79  for(int x = start; x < end; x++){
80  iter = counter.find((long)x);
81  if(iter != counter.end() && iter->second > 0.0){
82  double nx = iter->second;
83  double mean = x+1.0;
84  double sd = sqrt(2*((double)(x+1.0))*C/nx);
85  double px2 = 0.5*(1 + _erf((((double)k+1.5) - mean))/ (sd*sqrt(2.0)));
86  double px1 = 0.5*(1 + _erf((((double)k+0.5) - mean))/ (sd*sqrt(2.0)));
87  assert(nx > 0.0);
88  assert(mean > 0.0);
89  assert(sd > 0.0);
90  sum[k] += nx*(px2 - px1);
91  }
92  }
93  sum[k] = sum[k]/N;
94  total = total+ sum[k];
95  }
96  double epsilon = 1e-5;
97  total = total/(1 - max*epsilon);
98 
99  DoubleVector prob;
100  prob.resize(max+1);
101  for (int k = 1; k <= max; k++){
102  prob[k] = epsilon + sum[k]/total;
103  }
104 
105  ProbabilisticModelParameters pars;
106  pars.add("probabilities", ProbabilisticModelParameterValuePtr (new DoubleVectorParameterValue(prob)));
107  pars.add("alphabet", alpha->getParameterValue());
108  DiscreteIIDModelPtr result =
109  DiscreteIIDModelPtr(new DiscreteIIDModel());
110  result->initialize(pars);
111  return result;
112  }
