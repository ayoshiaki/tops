123  void InhomogeneousMarkovChain::removeSequenceFromModel(const Sequence & s, int phase){
124  for(int i = 0; i < (int)s.size()-1; i++)
125  phase = mod(phase + 1, maximumTimeValue()+1);
126  ContextTreeNodePtr c = _context_trees[phase]->getRoot();
127  ContextTreeNodePtr p;
128  int j;
129  for(j = s.size()-2; j >=0; j--){
130  if(c->isLeaf())
131  break;
132  p = c;
133  c = c->getChild(s[j]);
134  if(c == NULL)
135  {
136  c = p;
137  break;
138  }
139  }
140  for(; j >= 0; j--) {
141  for(int l = 0; l < (int)alphabet()->size(); l++) {
142  ContextTreeNodePtr n = _context_trees[phase]->createContext();
143  DoubleVector prob;
144  for(int i = 0; i < (int)n->alphabet_size(); i++)
145  prob.push_back(exp(c ->getDistribution()->log_probability_of(i)));
146  DiscreteIIDModelPtr distr = DiscreteIIDModelPtr(new DiscreteIIDModel(prob));
147  n->setDistribution(distr);
148  c->setChild(n, l);
149  }
150  c = c->getChild(s[j]);
151  }

160  sum += exp(current->getDistribution()->log_probability_of(symbol));
161  }

163  {

240  void InhomogeneousMarkovChain::initialize(const ProbabilisticModelParameters & parameters) {
241  ProbabilisticModelParameterValuePtr symbols = parameters.getMandatoryParameterValue("alphabet");
242  ProbabilisticModelParameterValuePtr position_specific_model_names = parameters.getMandatoryParameterValue("position_specific_distribution");
243  ProbabilisticModelParameterValuePtr p = parameters.getMandatoryParameterValue("phased");
244 
245  bool is_phased = (p->getInt() == 1);
246 
247 
248  AlphabetPtr alphabet = AlphabetPtr(new Alphabet());
249  alphabet->initializeFromVector(symbols->getStringVector());
250  std::vector <ContextTreePtr> position_specific_trees;
251  StringVector model_names = position_specific_model_names->getStringVector();
252  position_specific_trees.resize(model_names.size());
253  for(int i = 0; i < (int)model_names.size(); i++)
254  {
255  ProbabilisticModelParameters vlmc;
256  ProbabilisticModelParameterValuePtr probs = parameters.getMandatoryParameterValue(model_names[i]);
257  if (probs == NULL)
258  {
259  std::cerr << "ERROR: Could not find " << model_names[i] << std::endl;
260  }
261 
262  vlmc.add("alphabet", symbols);
263  vlmc.add("probabilities", probs);
264 
265  VariableLengthMarkovChainCreator vlmcCreator;
266  VariableLengthMarkovChainPtr mc = vlmcCreator.createVLMC(vlmc);
267  position_specific_trees[i] = mc->getTree();
268  }
269  phased(is_phased);
270  setAlphabet(alphabet);
271  setPositionSpecificDistribution(position_specific_trees);
272  }
