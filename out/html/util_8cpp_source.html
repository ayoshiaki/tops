29  /* Epanechnikov kernel */
30  double epanechnikov(double x, double h){
31  double a = h * sqrt(5.0);
32  double absx = fabs(x);
33  if(absx < a ) {
34  return (3.0/4.0) * (1.0 - (absx/a)*(absx/a))/a;
35  } else {
36  return 0.0;
37  }
38  }

51  void band_den_bin(int n, int nb, double *d, const DoubleVector &x, DoubleVector &cnt)
52  {
53  int i, j, nn = n;
54  int ii, jj, iij;
55  double xmin, xmax, rang, dd;
56  for (i = 0; i < nb; i++)
57  cnt.push_back(0);
58  xmin = xmax = x[0];
59  for (i = 1; i < nn; i++) {
60  xmin = _min(xmin, x[i]);
61  xmax = _min(xmax, x[i]);
62  }
63  rang = (xmax - xmin) * 1.01;
64  *d = dd = rang / (nb);
65  for (i = 1; i < nn; i++) {
66  ii = (int)(x[i] / dd);
67  for (j = 0; j < i; j++) {
68  jj = (int)(x[j] / dd);
69  iij = abs9((ii - jj));
70  cnt[iij]++;
71  }
72  }

74  void band_phi6_bin(int n, int nb, double d, DoubleVector &x, double h, double *u)
75  {
76  int i, nn = n, nbin = nb;
77  double delta, sum, term;
78  sum = 0.0;
79  for (i = 0; i < nbin; i++) {
80  delta = i * (d) / (h);
81  delta *= delta;
82  if (delta >= 1000) break;
83  term = exp(-delta / 2) *
84  (delta * delta * delta - 15 * delta * delta + 45 * delta - 15);
85  sum += term * x[i];
86  }
87  sum = 2 * sum - 15 * nn; /* add in diagonal */
88  *u = sum / (nn * (nn - 1) * pow(h, 7.0) * sqrt(2 * M_PI));
89  }

92  {

98  delta = i * (d) / (h);

126  /* quantile */
127  double quantile (DoubleVector data, double q){
128  int low_index = (int)floor(q * ((double)data.size()-1));
129  int high_index = (int)ceil(q * ((double)data.size()-1));
130  double h = q * ((double)data.size() -1) - (double)low_index;
131  sort(data.begin(), data.end());
132  return (1-h)*data[low_index] + h * data[high_index];
133  }

341  double sj_bandwidth(const DoubleVector &data){
342  double n = (double) data.size();
343  int nb = 1000;
344  double d = 1.0;
345  double variance = var(data);
346  DoubleVector count;
347  band_den_bin ((int)n, nb, &d, data, count);
348  double scale = _min(std::sqrt(variance), iqr(data)/1.349);
349  double b = 1.23 * scale * pow(n, (-1.0/9.0));
350  double c1 = 1.0 / (2.0*sqrt(M_PI) * n);
351  double td;
352  band_phi6_bin((int)n, (int)count.size(), d, count, b, &td);
353  td = -td;
354  if(td < 0 || td != td){
355  // cerr << "sj_bandwidth (WARNING). Dataset very sparse !!!\n" << endl;
356  return 0.001;
357  }
358  double sdh ;
359  band_phi4_bin((int)n, (int)count.size(), d, count, pow(2.394/(n*td), (1.0/7.0)), &sdh);
360  return pow((c1/sdh), 1.0/5.0);
361  }
