35 ProbabilisticModelPtr TrainFixedLengthMarkovChain::create(
36  ProbabilisticModelParameters & parameters, double & loglikelihood,
37  int & sample_size) const {
38  ProbabilisticModelParameterValuePtr orderpar =
39  parameters.getMandatoryParameterValue("order");
40  ProbabilisticModelParameterValuePtr trainpar =
41  parameters.getMandatoryParameterValue("training_set");
42  ProbabilisticModelParameterValuePtr alphapar =
43  parameters.getMandatoryParameterValue("alphabet");
44  ProbabilisticModelParameterValuePtr pseudocountspar = parameters.getOptionalParameterValue("pseudo_counts");
45  ProbabilisticModelParameterValuePtr aprioripar = parameters.getOptionalParameterValue("apriori");
46  ProbabilisticModelParameterValuePtr weightspar = parameters.getOptionalParameterValue("weights");
47  std::map <std::string, double> weights;
48  if(weightspar != NULL) {
49  readMapFromFile(weights, weightspar->getString());
50  }
51 
52 
53 
54  double pseudocounts = 0;
55  ProbabilisticModelPtr apriori;
56  if(aprioripar != NULL)
57  {
58  ProbabilisticModelCreatorClient c;
59  apriori = c.create(aprioripar->getString());
60  }
61  if(pseudocountspar != NULL)
62  pseudocounts = pseudocountspar->getDouble();
63 
64 
65  if ((trainpar == NULL) || (alphapar == NULL) || (orderpar == NULL)) {
66  std::cerr << help() << std::endl;
67  exit(-1);
68  }
69  AlphabetPtr alphabet = AlphabetPtr(new Alphabet());
70  alphabet ->initializeFromVector(alphapar->getStringVector());
71  SequenceEntryList sample_set;
72  readSequencesFromFile(sample_set, alphabet, trainpar->getString());
73  ContextTreePtr tree = ContextTreePtr(new ContextTree(alphabet));
74 
75  if(apriori != NULL ){
76  tree->initializeCounter(sample_set, orderpar->getInt(), 0, weights);
77  tree->normalize(apriori, pseudocounts);
78  } else {
79  tree->initializeCounter(sample_set, orderpar->getInt(), pseudocounts, weights);
80  tree->normalize();
81  }
82 
83  VariableLengthMarkovChainPtr m = VariableLengthMarkovChainPtr(
84  new VariableLengthMarkovChain(tree));
85  m->setAlphabet(alphabet);
86  loglikelihood = 0.0;
87  sample_size = 0;
88  for (int i = 0; i < (int) sample_set.size(); i++) {
89  loglikelihood
90  += m->evaluate((sample_set[i]->getSequence()), 0, (sample_set[i]->getSequence()).size() - 1);
91  sample_size += (sample_set[i]->getSequence()).size();
92  }
93 
94  return m;
95 
96 }
