67  double calculate_normalizer (int skip_length,int skip_offset, int max_length, std::map<std::string, double > & counter, int alphabet_size)
68  {
69  int npatterns_differ_1 = 0;
70  npatterns_differ_1 = (alphabet_size - 1) * (max_length - skip_length);
71  if(skip_length < 0)
72  npatterns_differ_1 = (alphabet_size - 1) * (max_length );
73  double sum = 0.0;
74  std::map<std::string, double>::const_iterator it;
75  std::map<std::string, double>::const_iterator it2;
76  for(it = counter.begin(); it != counter.end(); it++)
77  {
78  sum += it->second;
79 
80  int diff = 0;
81  int np_differ_1 = 0;
82 
83  for(it2 = counter.begin(); it2 != counter.end(); it2++)
84  {
85  std::string a = it->first;
86  std::string b = it2->first;
87  for(int i = 0; i < max_length; i++)
88  if((i >= skip_offset) && (i <= skip_offset+skip_length)){
89  if(a[i] != b[i])
90  diff+=2;
91  }else if(a[i] != b[i])

93

95  //sum += it2->second * 0.001;

97  }
98  }
99  sum += 0.001*it->second*(npatterns_differ_1 - np_differ_1);
100  }
