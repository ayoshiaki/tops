93  double SimilarityBasedSequenceWeighting::evaluate(const Sequence & s, unsigned int begin, unsigned int end) const {
94  if (end >= s.size())
95  return -HUGE;
96  int length = (_counter.begin()->first).size();
97  std::stringstream qstream;
98  for(int i = begin ; (i <= (int)end) && (i < (int) (begin + length)) ; i++){
99  qstream << alphabet()->getSymbol(s[i])->name();
100  }
101  std::string q = qstream.str();
102  int psize = 0;
103  std::map<std::string, double>::const_iterator it;
104  double sum = 0;
105  for(it = _counter.begin(); it != _counter.end(); it++)
106  {
107  std::string q2 = it->first;
108  psize = q2.size();
109  int diff = 0;
110  if(q.size () != q2.size())
111  return -HUGE;
112  bool valid = true;
113  for(int i = 0; i < (int)q2.size(); i++)
114  {
115  if((i >= _skip_offset) && (i < _skip_offset+_skip_length)){
116  if(q[i] != _skip_sequence[i-_skip_offset]){
117  valid = false;
118  break;
119  }
120  }else if(q[i] != q2[i])
121  diff++;
122  }
123  if(!valid)
124  return -HUGE;
125  if(diff == 1){
126  sum += 0.001 * it->second;
127  } else if (diff==0){
128  sum += it->second;
129  }
130  }
131  if(close(sum , 0.0, 1e-10))
132  return -HUGE;
133  return log(sum/(_normalizer));
134  }
