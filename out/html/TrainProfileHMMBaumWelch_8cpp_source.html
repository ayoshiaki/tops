78 void TrainProfileHMMBaumWelch::create_emiss_file(string fastafile, string datetime) const {
79  string line, sequencia;
80  vector<string> seq;
81  vector<string> seq_names;
82  ifstream myfile(fastafile.c_str());
83 
84  if (myfile.is_open()) {
85  while (!myfile.eof()) {
86  getline(myfile, line);
87  if (match_expression(line, ">.*")) {
88  if (seq_names.size() != 0) {
89  seq.push_back(sequencia);
90  sequencia = "";
91  }
92  seq_names.push_back(line.substr(1, line.length()));
93  } else {
94  sequencia += line;
95  }
96  }
97  if (seq_names.size() != 0) {
98  seq.push_back(sequencia);
99  sequencia = "";
100  }
101  myfile.close();
102 
103  }
104 
105  string emiss_filename = ".emiss_" + datetime + ".txt";
106 
107  ofstream emiss;
108  emiss.open(emiss_filename.c_str());
109  for (unsigned int i = 0; i < seq.size(); i++) {
110  emiss << seq_names[i] << "_emiss: ";
111  for (unsigned int j = 0; j < seq[i].size(); j++) {
112 
113  if (seq[i][j] != '-')
114  emiss << seq[i][j] << " ";
115 
116  }
117  emiss << endl;
118 
119  }
120  emiss.close();
121 
122 }
123 
124 }
125 ;
