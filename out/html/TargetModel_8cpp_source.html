49  double TargetModel::evaluate(const Sequence & s, unsigned int begin, unsigned int end) const {
50  if (end >= s.size())
51  return -HUGE;
52  Sequence subseq(end - begin + 1);
53  int k = 0;
54  for(int i = begin ; i <= (int)end; i++){
55  subseq[k] = s[i];
56  k++;
57  }
58  SequenceEntryList samples;
59  SequenceEntryPtr entry = SequenceEntryPtr(new SequenceEntry(alphabet()));
60  entry->setSequence(subseq);
61  samples.push_back(entry);
62  TrainDiscreteIIDModelPtr trainingAlgorithm = TrainDiscreteIIDModelPtr(new TrainDiscreteIIDModel());
63  ProbabilisticModelPtr m = trainingAlgorithm->train(samples, alphabet());
64  return m->evaluate(s, begin, end);
65  }
